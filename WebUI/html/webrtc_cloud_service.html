WebRTC Cloud Service Configuration Example
            
<script type="application/javascript">
    window.GSWebRTCSettings = {
        custom: {
            position: {
                bottom: 0,
                right: 0
                // top: "20%",
                // left: "30%"
            },
            language: "en-US" // zh-CN 
        },
        server: {
            url: "https://ucmwebrtc.ipvideotalk.com",
            path: "GSWebRTC",
            version: "1.1"
        },
        SIP: {
            realm: "Grandstream",
            impi: "Anonymous",
            impu: "sip:webrtc_anonymous@anonymous.invalid",
            display_name: "Anonymous",
            phone_number: "WebRTC",
            cloud_account: "Your Cloud Account"
        }
    };
    var server = GSWebRTCSettings.server,
        url = server.url,
        path = server.path,
        version = server.version;

    var myScript = document.createElement("script");
        myScript.type = "text/javascript";
        myScript.src = (url + "/" +
                        path + "/" +
                        version + "/" +
                        "webRTC.js");

    if(typeof jQuery == "undefined") {
        var jQueryScript = document.createElement("script");
            jQueryScript.type = "text/javascript";
            jQueryScript.src = (url + "/" +
                                path + "/" +
                                version + "/" +
                                "jquery.js");
        document.body.appendChild(jQueryScript);

        jQueryScript.onload = function () {
            document.body.appendChild(myScript);
        }

        jQueryScript.onerror = function() {
            document.body.appendChild(jQueryScript);
        }
    } else {
        document.body.appendChild(myScript);

        myScript.onerror = function() {
            document.body.appendChild(myScript);
        }
    }
</script>
